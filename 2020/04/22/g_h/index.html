简易购物车代码

2020-04-22 289 words
Vue

如下

<template>
<div>
    <table>
    <tr>
    <td>名称</td>
    <td>数量</td>
    <td>价格</td>
    </tr>
    <tr v-for="(itemone,indexone) in tlist " :key="indexone">
    <td>
        {{ itemone.text }}
    </td>

    <td>
        <button @click="numchange(indexone,'-')" >-</button>
            {{itemone.num}}
        <button @click="numchange(indexone,'+')">+</button>
    </td>

    <td>
        {{ itemone.price }}
    </td>
    </tr>

</table>
    总价格:{{ totalprice() }}
</div>
</template>

<script>
export default{
// 定义数据
    data(){
    return{
    tlist:[{text:'汽车',num:1,'price':100},{text:'化妆品',num:1,'price':100,},{text:'衣服',num:1,'price':100},{text:'书籍',num:1,'price':100}],
        }
    },
    //自定方法
    methods:{
        // 计算总的价格 
        totalprice:function(){
            let total =0;
            for(var i=0,l=this.tlist.length;i<l;i++){
                total += (this.tlist[i].num * this.tlist[i].price);
            },
            return total;
        },
        // 自动清除操作
        delete_one:function(indexone){
            this.tlist.splice(indexone,1);
        },

        // 购物车的加减操作
        numchange:function(indexone,type){
            if(type=="+"){
            this.tlist[indexone].num ++
            }else{
                if (this.tlist[indexone].num > 1) {
                    this.tlist[indexone].num --
                }else {
                    this.delete_one(indexone);
                    
                }	
            } 
        },
    },
}
</script>
