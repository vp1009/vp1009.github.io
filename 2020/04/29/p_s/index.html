图片压缩

#压缩 两种压缩模式，png不适合压缩，jpg适合
import cv2
#读图
img = cv2.imread('./dingding.png')
#开始压缩 png压缩等级清晰0-9模糊
cv2.imwrite('./dingding1.png',img,[cv2.IMWRITE_PNG_COMPRESSION,5])

#jpg压缩等级清晰0-100模糊 
cv2.imwrite('./dingding1.jpg',img,[cv2.IMWRITE_JPEG_QUALITY,50])

加水印

#绘画库
from PIL import ImageDraw
#字体库
from PIL import ImageFont
#图片库
from PIL import Image

#加水印
img = Image.open('.//static/upload/'+myfile_name)
# 获取宽高
width, height = img.size
# 根据图片大小定义logo大小
text = '哈哈哈'
#字体
font = ImageFont.truetype(font='C:\Windows\Fonts\FZXKJW.TTF',size=35)
# 获取字体宽高
font_width, font_height = font.getsize(text)
# 使用画笔
draw = ImageDraw.Draw(img)
# 写入文本 参数（（x,y），水印文字，颜色，字体）
draw.text((width-font_width,height-font_height), text,fill=(76,234,124,180), font=font)
